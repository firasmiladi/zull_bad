274    /**
275     * Given a command, process (that is: execute) the command.
276     * If this command ends the game, true is returned, otherwise false is
277     * returned.
278     * @param pString user command 
279     */
280    public void interpretCommand( final String pString) 
281    {
282        if (this.aCount==0 || this.aLoser){
283            this.aGui.boutonvisibility(false, false, false, false, false, false);
284            return;
285        }
286        this.aGui.println( "\n > " + pString );
287
288        Command vCommand = this.aParser.getCommand(pString);
289        CommandWord vCommandWord = vCommand.getCommandWord();
290
291        switch ( vCommandWord ){ 
292            case HELP: this.printHelp(); 
293            break; 
294            case GO:this.goRoom( vCommand );
295            break; 
296            case EAT: this.eat( vCommand ); 
297            break; 
298            case LOOK: this.look(); 
299            break; 
300            case QUIT: this.quit(); 
301            break; 
302            case TEST: this.test(vCommand); 
303            break;
304            case TAKE: this.take(vCommand); 
305            break;
306            case BACK:this.Back();
307            break;
308            case DROP: this.drop(vCommand); 
309            break; 
310            case ITEMS: this.printInventory(); 
311            break;
312            case LOAD: this.load(vCommand); 
313            break;
314            case TELEPORT: this.teleport(); 
315            break;
316            case LOCK: this.lockDoor(vCommand); 
317            break;
318            case UNLOCK: this.unlockDoor(vCommand); 
319            break;
320            case UNKNOWN: this.aGui.println( "I don't know what you mean..." ); 
321            break;
322            case ALEA: this.alea(vCommand); 
323            break;
324        }
325
326        if (--this.aCount==0){quit();}
327    }

431    /**
432     * give,choose directions/exits 
433     * @param pCommand desired direction 
434     */
435    private void goRoom( final Command pCommand ) 
436    {
437        if ( ! pCommand.hasSecondWord() ) {
438            // if there is no second word, we don't know where to go...
439            this.aGui.println( "Go where?" );
440            return;
441        }
442
443        String vDirection = pCommand.getSecondWord();
444
445        // Try to leave current room.
446        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit( vDirection );
447        Room nextDoor = aPlayer.getCurrentRoom().getDoor(vDirection);
448        
449        
450        if (nextDoor != null)
451        {
452            if (aPlayer.getCurrentRoom().getLocked(vDirection))
453            {
454                this.aGui.println("The door is locked! \n You need to unlocked it \n First you need to find the key !");
455                return;
456            }
457            else
458            {
459                
460                this.aBackRooms.push(this.aPlayer.getCurrentRoom());
461                aPlayer.setCurrentRoom(nextDoor);
462
463                //this.aGui.println(aPlayer.getCurrentRoom().getLongDescription());
464                this.printLocationInfo();   
465                if ( this.aPlayer.getCurrentRoom().getImageName() != null )
466                {
467                    this.aGui.showImage( this.aPlayer.getCurrentRoom().getImageName() );
468                    return;
469                }
470            }
471
472        }
473        else if ( vNextRoom != null )
474        {
475            this.aBackRooms.push(this.aPlayer.getCurrentRoom()); 
476            this.aPlayer.setCurrentRoom(vNextRoom);
477             if(this.aPlayer.getCurrentRoom()==this.aRooms.get("home")&&this.aPlayer.checkWinner())
478             { this.aGui.println("Congruatulation! \n you are a Winner! \n you return to your home safe and you saved your family! ");
479                 //return;
480        }   else if(this.aPlayer.getCurrentRoom()==this.aRooms.get("home")&&!this.aPlayer.checkWinner()&&this.aLife>0)
481        { 
482          this.aGui.boutonvisibility(false, true, false, false, false, false);
483            this.aGui.println("No!\n you're in your home while your family is in danger \n you can't abandon Your Family \n you have just "+this.aLife+" chance to rescue your family and return to your home safe" );
484          this.aLife--;
485          //return;
486        }   else if(this.aPlayer.getCurrentRoom()==this.aRooms.get("home")&&!this.aPlayer.checkWinner()&& this.aLife==0)
487        { 
488          this.aGui.println("You lost! \n you return to your house and you did abandon Your Family!" );
489          quit();
490          this.aLoser=true;
491          //return;
492        }else printLocationInfo();
493            
494            
495            if ( this.aPlayer.getCurrentRoom().getImageName() != null )
496                this.aGui.showImage( this.aPlayer.getCurrentRoom().getImageName() );
497
498        }
499    }
