238    /**
239     * Given a command, process (that is: execute) the command.
240     * If this command ends the game, true is returned, otherwise false is
241     * returned.
242     * @param pString user command 
243     */
244    public void interpretCommand( final String pString) 
245    {
246        if (this.aCount==0){return;}
247        this.aGui.println( "\n > " + pString );
248        
249        Command vCommand = this.aParser.getCommand(pString);
250        CommandWord vCommandWord = vCommand.getCommandWord();
251        
252        switch ( vCommandWord ){ 
253            case HELP: this.printHelp(); 
254            break; 
255            case GO: this.goRoom( vCommand );
256            break; 
257            case EAT: this.eat( vCommand ); 
258            break; 
259            case LOOK: this.look(); 
260            break; 
261            case QUIT: this.quit(); 
262            break; 
263            case TEST: this.test(vCommand); 
264            break;
265            case TAKE: this.take(vCommand); 
266            break;
267            case BACK: this.Back();
268            break;
269            case DROP: this.drop(vCommand); 
270            break; 
271            case ITEMS: this.printInventory(); 
272            break;
273            case LOAD: this.load(vCommand); 
274            break;
275            case TELEPORT: this.teleport(); 
276            break;
277            case LOCK: this.lockDoor(vCommand); 
278            break;
279            case UNLOCK: this.unlockDoor(vCommand); 
280            break;
281            case UNKNOWN: this.aGui.println( "I don't know what you mean..." ); 
282            break;
283        }
284            
285        if (--this.aCount==0){quit();}
286    }

312    /**
313     * take item
314     * @param pCommand item 's noun
315     */
316    private void take(final Command pCommand) {
317            if ( !pCommand.hasSecondWord() ) {
318                this.aGui.println("\n" + "what do you want to take?" + "\n");
319                return;
320            }
321            String vCommandWord = pCommand.getSecondWord();
322            Item vItem = this.aPlayer.getCurrentRoom().getRoomItemlist().getItemList(vCommandWord);
323            if(vItem==null){
324                this.aGui.println("\n" + "there is no item" + "\n");
325                return;
326            }
327            if(vItem.getWeightItem()+this.aPlayer.getWeightInventory()<=this.aPlayer.getOneRepMax() && vItem.getPriceItem()<=this.aPlayer.getwallet()) {
328                String vCommand = pCommand.getSecondWord();
329                this.aGui.println(this.aPlayer.takeString(vCommand));
330                this.aGui.println(this.aPlayer.getCurrentRoom().getRoomItemlist().getItemListStringRoom());
331                
332            }
333            else if(vItem.getWeightItem()+this.aPlayer.getWeightInventory()<this.aPlayer.getOneRepMax() && vItem.getPriceItem()>this.aPlayer.getwallet()) {
334                this.aGui.println("it's to expensive man!\n you don't have enough money to pick this item");
335                return;
336        
337            }
338        
339            else if(vItem.getWeightItem()+this.aPlayer.getWeightInventory()>this.aPlayer.getOneRepMax() && vItem.getPriceItem()<this.aPlayer.getwallet()) {
340                this.aGui.println("it's to heavy man!\n you don't have enough power to pick this item");
341                return;
342        
343            }
344            else if(vItem.getWeightItem()+this.aPlayer.getWeightInventory()>this.aPlayer.getOneRepMax() && vItem.getPriceItem()>this.aPlayer.getwallet()) {
345                this.aGui.println("it's to heavy and to expensive man!\n Try to pick another item");
346                return;
347

383        Room nextDoor = aPlayer.getCurrentRoom().getDoor(vDirection);
384        
385        if (nextDoor != null)
386        {
387            if (aPlayer.getCurrentRoom().getLocked(vDirection))
388            {
389                this.aGui.println("The door is locked! \n You need to unlocked it \n First you need to find the key !");
390                return;
391            }
392            else
393            {
394                aPlayer.setCurrentRoom(nextDoor);
395                this.aBackRooms.push(this.aPlayer.getCurrentRoom());

400                        this.aGui.showImage( this.aPlayer.getCurrentRoom().getImageName() );

402                    }
403                System.out.println("testrrr");
404             }
405             
406        }
407        else if ( vNextRoom != null )

409            this.aBackRooms.push(this.aPlayer.getCurrentRoom());
