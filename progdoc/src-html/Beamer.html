010public class Beamer extends Item{
011    private Room aSavedRoom;
012    private boolean aLoaded;
013    
014    /**
015     * Constructor - initialise the command words.
016     * 
017     * @param pNoun the object noun .
018     * @param pDescription the object description 
019     * @param pPrice the object price 
020     * @param pWeight the object weight 
021     */
022    public Beamer(final String pNoun, final String pDescription, final double pPrice,final double pWeight){
023        super(pNoun, pDescription, pPrice, pWeight);
024        this.aLoaded = false;
025    }
026    
027    /**
028     * @return The Saved Room.
029     */
030    public Room getSavedRoom(){
031        return this.aSavedRoom;
032    }
033    
034    /**
035     * set Saved Room.
036     * @param pSavedRoom desired room 
037     */
038    public void setSavedRoom(final Room pSavedRoom){
039        this.aSavedRoom = pSavedRoom;
040    }
041    /**
042     * @return The beamer is Loaded or not 
043     */
044    public boolean isLoaded(){
045        return this.aLoaded;
046    }
047    /**
048     * set Loaded the beamer.
049     * @param pLoaded Loaded or not 
050     */
051    public void setLoaded(final boolean pLoaded){
052        this.aLoaded = pLoaded;
053    }
054}
