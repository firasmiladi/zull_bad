001import java.util.Random;
002import java.util.ArrayList;
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class is used to generate randomly a room.
008 * 
009 * @author Firas miladi  
010 * @version 30/04/2020
011 */
012public class RoomRandomizer { 
013    private Random aRandom;
014    private ArrayList<Room> aArray; 
015     /**
016     * Constructor
017     * @param pRoom : get an array of all rooms
018     */
019    public RoomRandomizer(final ArrayList pRoom)
020    {
021        this.aRandom = new Random();
022        this.aArray = new ArrayList<Room>(pRoom);
023    } 
024    /**
025     * @return a room choosen randomly  .
026     */
027    public Room nextRoom()
028    {
029        int vRandom = aRandom.nextInt(aArray.size());
030        return this.aArray.get(vRandom);
031    }
032
033}
