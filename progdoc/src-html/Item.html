001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class is used to trait Items .
006 * 
007 * @author Firas miladi  
008 * @version 30/04/2020
009 */
010public class Item
011{
012    private String aNoun;
013    private String aDescription;
014    private double aPrice;
015    private double aWeight;
016    private boolean aNonPlayer;
017    /**
018     * Constructor
019     * @param pNoun :get Item's noun
020     * @param pDescription : get Item's Description .
021     * @param pPrice : get Item's Price .
022     * @param pWeight : get Item's weight .
023     */
024    public Item(final String pNoun, final String pDescription, final double pPrice, final double pWeight)
025    {
026        this(pNoun,pDescription);
027        this.aPrice        = pPrice;
028        this.aWeight       = pWeight;
029        this.aNonPlayer= false;
030    }
031    
032    /**
033     * Constructor
034     * @param pNoun :get Item's noun
035     * @param pDescription : get Item's Description .
036     * 
037     */
038    public Item(final String pNoun, final String pDescription)
039    {
040        this.aNoun        = pNoun;
041        this.aDescription = pDescription;
042        this.aNonPlayer   = true;
043    }
044    
045    /**
046     * @return if the item is a non-player   .
047     */
048    public boolean getNonPlayer(){return this.aNonPlayer;}
049    
050    /**
051     * @return get item noun.
052     */
053    public String getNounItem(){return this.aNoun;}
054      
055    /**
056     * @return get item description for gold and non-player.
057     */
058    public String getDescriptionItem(){return this.aNoun+" || "+this.aDescription ;}
059    
060    /**
061     * @return get Item's Description .
062     */
063    public String getDescriptionItemString(){return this.aDescription + " || price : "+ this.aPrice +" Yuan || Weight : "+ this.aWeight +" Kg";}
064    
065    /**
066     * @return get full Item's Description .
067     */
068   public String getLongDescriptionItem(){return this.aNoun+" || "+ this.getDescriptionItemString();}
069
070    /**
071     * @return get Item's price .
072     */
073    public double getPriceItem(){return this.aPrice;}
074    
075    /**
076     * @return get Item's weight .
077     */
078    public double getWeightItem(){return this.aWeight;}
079    
080}
