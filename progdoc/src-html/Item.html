001
002public class Item
003{
004    private String aNoun;
005    private String aDescription;
006    private double aPrice;
007    private double aWeight;
008    
009    /**
010     * Constructor
011     * @param pNoun :get Item's noun
012     * @param pDescription : get Item's Description .
013     * @param pPrice : get Item's Price .
014     * @param pWeight : get Item's weight .
015     */
016    public Item(final String pNoun, final String pDescription, final double pPrice, final double pWeight)
017    {
018        this.aNoun        = pNoun;
019        this.aDescription = pDescription;
020        this.aPrice        = pPrice;
021        this.aWeight       = pWeight;
022    }
023    
024    /**
025     * @return get item noun.
026     */
027    public String getNounItem(){return this.aNoun;}
028      
029    /**
030     * @return get item description for gold.
031     */
032    public String getDescriptionItem(){return this.aNoun+" || "+this.aDescription ;}
033    
034    /**
035     * @return get Item's Description .
036     */
037    public String getDescriptionItemString(){return this.aDescription + " || price : "+ this.aPrice +" Yuan || Weight : "+ this.aWeight +" Kg";}
038    
039    /**
040     * @return get full Item's Description .
041     */
042   public String getLongDescriptionItem(){return this.aNoun+" || "+this.aDescription + " || price : "+ this.aPrice +" Yuan || Weight : "+ this.aWeight +" Kg";}
043
044    /**
045     * @return get Item's price .
046     */
047    public double getPriceItem(){return this.aPrice;}
048    
049    /**
050     * @return get Item's weight .
051     */
052    public double getWeightItem(){return this.aWeight;}
053    
054}
