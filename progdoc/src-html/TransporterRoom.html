001import java.util.HashMap;
002import java.util.ArrayList;
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class is used to trait Transporter Room .
008 * 
009 * @author Firas miladi  
010 * @version 30/04/2020
011 */
012public class TransporterRoom extends Room{
013    private HashMap<String,Room> aRooms;
014    private ArrayList<Room> aArrayInRooms;
015    private RoomRandomizer aRandomRoom;
016    private String aAleaString;
017    /**
018     * Constructor
019     * @param pDescription :get the room Description
020     * @param pImage : get the room image .
021     * @param pRooms : get table of rooms .
022     */
023    public TransporterRoom(final String pDescription,final String pImage, final HashMap<String,Room> pRooms)
024    {
025        super(pDescription, pImage);
026        this.aRooms = pRooms;
027        this.aArrayInRooms = new ArrayList<Room>(this.aRooms.values());
028        this.aRandomRoom = new RoomRandomizer(this.aArrayInRooms);
029        this.aAleaString = null ;
030    }
031    
032    /**
033     * @param pDirection direction exit
034     * @return get a room randomly .
035     */
036    @Override 
037    public Room getExit(final String pDirection)
038    {
039        if(this.aAleaString==null){return this.findRandomRoom();}
040        else{return aRooms.get(aAleaString);}
041    }
042    
043    /**
044     * choose the destination (TEST MODE ONLY)
045     * @param pRoom room name
046     */
047    public void setAlea(final String pRoom)
048    {
049        this.aAleaString = pRoom ;
050    }
051    
052    /**
053     * @return get a room randomly .
054     */
055    private Room findRandomRoom () {
056        return this.aRandomRoom.nextRoom();
057    }
058} // TransporterRoom
