001import java.util.HashMap;
002
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class holds an enumeration table of all command words known to the game.
008 * It is used to recognise commands as they are typed in.
009 *
010 * @author  Firas Miladi
011 * 
012 * @version 10.04.2020
013 */
014public class CommandWords
015{
016    // a constant array that will hold all valid command words
017    private HashMap<String, CommandWord> avalidCommands;
018
019    /**
020     * Constructor - initialise the command words.
021     */
022    public CommandWords()
023    {
024        this.avalidCommands = new HashMap<String, CommandWord>();
025        for(CommandWord vcommand : CommandWord.values()) {
026            if(vcommand != CommandWord.UNKNOWN) {
027                this.avalidCommands.put(vcommand.toString(), vcommand);
028            }
029        }
030    }
031    
032    /**
033     * Check whether a given String is a valid command word. 
034     * @return true if a given string is a valid command,
035     * false if it isn't.
036     * @param pString command input
037     */
038    public boolean isCommand( final String pString )
039    {
040        return this.avalidCommands.containsKey(pString);
041    } // isCommand()
042    
043    /**
044     * Print all valid commands to System.out.
045     * @return all valid commands
046     */
047    public String showAll() 
048    {
049        StringBuilder returnString = new StringBuilder("");
050        
051        for ( String vS : avalidCommands.keySet() )
052                returnString.append( " " + vS );
053        return returnString.toString();
054    }
055    
056     /**
057     * Find the CommandWord associated with a command word.
058     * @param pCommandWord The word to look up.
059     * @return The CommandWord correspondng to commandWord, or UNKNOWN
060     *         if it is not a valid command word.
061     */
062    public CommandWord getCommandWord(String pCommandWord)
063    {
064        CommandWord vcommand = this.avalidCommands.get(pCommandWord);
065        if(vcommand != null) {
066            return vcommand;
067        }
068        else {
069            return CommandWord.UNKNOWN;
070        }
071    }
072}
073 // CommandWords
