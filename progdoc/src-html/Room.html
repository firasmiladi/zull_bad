001import java.util.HashMap;
002import java.util.Set;
003 /**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class holds a HashMap of all rooms and their exits of the game.
008 *
009 * @author  Miladi Firas
010 * 
011 * @version 2021.02.01
012 */
013public class Room
014{
015    private String aDescription ;
016    private HashMap<String, Room> aExits;
017    private String aImageName;
018    private Item aItem;
019    private ItemList alist;
020    private HashMap<String, Door> aDoors;
021
022    /**
023     * Constructor - to create a room . 
024     * 
025    * @param pDescription The room's description.
026    * @param pImage The room's image.
027    */
028    public Room(final String pDescription, final String pImage)
029    {
030        this.aDescription = pDescription;
031        this.aExits= new HashMap<String, Room> ();
032        this.aImageName = pImage;
033        this.alist= new ItemList ();
034        this.aDoors = new HashMap<String, Door>();
035    }//Room
036    
037    /**
038     * get the room's description.
039     * 
040     * @return the room's description.
041     */
042    public String getDescription()
043    {
044        return this.aDescription;
045    }//getDescription
046    
047    /**
048     * define the exits ,every direction steer to another room 
049     * @param pDirection desired direction 
050     * @param pNeighbor the correspondent room
051     */
052    public void setExits(final String pDirection,final Room pNeighbor )
053    {
054        this.aExits.put(pDirection, pNeighbor);
055    }//setExits
056    
057    /**
058     * set Door.
059     * @param pDirection noun 
060     * @param pNeighbor room 
061     * @param pLocked Locked or not 
062     */
063    public void setDoor(String pDirection, Room pNeighbor, boolean pLocked)
064    {
065        Door temp = new Door(pDirection, pNeighbor, pLocked);
066        this.aDoors.put(pDirection, temp);
067    }
068    
069    
070    /**
071     * get the new room . 
072     * @param pDirection desired direction
073     * @return new room .
074     */
075    public Room getExit(final String pDirection)
076    {
077        return this.aExits.get(pDirection);
078    }//getExit
079    
080    /**
081     * add anitem in the room . 
082     * @param pItem desired item
083     */
084    public void addItemroom  (Item pItem){
085        this.getRoomItemlist().addItemToTheList(pItem);
086    }
087
088    /**
089     * get the room's Itemlist . 
090     * @return Itemlist.
091     */
092    public ItemList getRoomItemlist()
093    {
094        return this.alist;
095    }//getRoomItemlist
096    
097    
098    /**
099     * get the Exit in a String.
100     * @return different exits in a String.
101     */
102    public String getExitString()
103    {
104        if (aExits.size()==0){return "there is no exits";}
105        else{
106        StringBuilder vReturnString = new StringBuilder("Exits: ");
107        
108        for ( String vS : aExits.keySet() )
109            vReturnString.append( " " + vS );
110        return vReturnString.toString();
111    }
112        
113        
114    }//getExitString
115    
116    /**
117     * get the Doors in a String.
118     * @return different doors in a String.
119     */
120    public String getDoorString()
121    {
122        String vReturnString = "Doors:";
123        if (aDoors.size()==0){return "there is no door in this room";}
124        else{
125        Set<String> keys = aDoors.keySet();
126        for(String vdoor : keys)
127            vReturnString += " " + vdoor;
128        return vReturnString;
129      }
130    }
131    
132    /**
133     * @return get Description of the room, the exits, and items .
134     */
135    public String getLongDescription()
136    {
137        return "You are "+ this.aDescription+"\n" +getExitString()+"\n" + getDoorString() +"\n"+ this.alist.getItemListStringRoom();
138    }//getLongDescription
139     
140    /**
141     * @return a string describing the room's image name
142     */
143     public String getImageName()
144     {
145         return this.aImageName;
146    }
147    
148    /**
149     * Return the room reached if we go from room A to B.
150     * If there is no room in that direction, return null.
151     * @param pDirection desired direction
152     * @return desired door
153     */
154    public Room getDoor(String pDirection)
155    {
156        Door tempDoor = this.aDoors.get(pDirection);
157        if (tempDoor != null)
158        {
159            return tempDoor.getNeighbor();
160        }
161        return null;
162    }
163    
164    /**
165     * @param pDirection desired direction
166     * @return locked or not.
167     */
168    public boolean getLocked(String pDirection)
169    {
170        return this.aDoors.get(pDirection).getLocked();
171    }
172    
173    
174    
175    /**
176     * @param pDirection desired direction
177     * @return The Actual Door.
178     * 
179     */
180    public Door getActualDoor(String pDirection)
181    {
182        return this.aDoors.get(pDirection);
183    }
184    
185}
