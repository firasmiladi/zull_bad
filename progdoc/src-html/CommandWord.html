001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds a table of all command words
006 * 
007 * @author Firas miladi  
008 * @version 20/04/2020
009 */
010public enum CommandWord
011{
012    // A value for each command word, plus one for unrecognised
013    // commands.
014    GO("go"), BACK("back") , HELP("help"), LOOK("look"), EAT("eat"), ALEA("alea"), TAKE("take"), DROP("drop"), TEST("test"), ITEMS("items"), UNKNOWN("?"), LOAD("load"), TELEPORT("teleport"), LOCK("lock"),UNLOCK("unlock"), QUIT("quit") ;
015    
016    private String aCommandString;
017    /**
018     * Initialise with the corresponding command word.
019     * @param pCommandString The command string.
020     */
021    CommandWord(final String pCommandString)
022    {this.aCommandString    =    pCommandString;}
023    
024    /**
025     * @return The command word as a string.
026     */
027    public String toString()
028    {
029        return this.aCommandString;
030    }
031} // CommandWord
