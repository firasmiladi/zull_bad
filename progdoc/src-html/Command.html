001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class is used to trait commands as they are typed in.
006 *
007 * @author  Miladi Firas
008 * 
009 * @version 2021.02.1
010 */
011 public class Command
012{
013    private CommandWord aCommandWord;
014    private String aSecondWord;
015    /**
016     * Constructor - initialise the command words.
017     * 
018     * @param pCommandWord commandWord The CommandWord. UNKNOWN if the command word
019     *                  was not recognised.
020     * @param  pSecond:second word of the command
021     */
022    public Command(final CommandWord pCommandWord ,final String pSecond) 
023    {
024        this.aCommandWord = pCommandWord;
025        this.aSecondWord = pSecond;
026    }//Command
027    
028    /**
029     * Return the command word (the first word) of this command.
030     * @return The command word.
031     */
032    public CommandWord getCommandWord()
033    {
034        return this.aCommandWord;
035    }//getCommandWord
036    
037    /**
038     * get the second word of the command
039     * @return the second word of the command from the user.
040     */
041    public String getSecondWord()
042    {
043        return this.aSecondWord;
044    }//getSecondWord
045    
046    /**
047     * Check if the command has a second word. 
048     * @return true if the command has a second  word,
049     * false if not.
050     */
051    public boolean hasSecondWord ()
052    {
053        return this.aSecondWord != null ; 
054    }//hasSecondWord
055    
056    /**
057     * Check if there is a command or not . 
058     * @return true if there is not a command
059     * false if there is .
060     */
061    public boolean isUnknown ()
062    {
063        return (this.aCommandWord  == this.aCommandWord .UNKNOWN); 
064    }//isUnknown
065}// Command
