001public class Door
002{
003    private boolean alocked;
004    private Room aNeighbor;
005    private String aDirection;
006    
007    /**
008     * Constructor for objects of class Door
009     * @param pDirection desared Direction
010     * @param pNeighbor next room
011     * @param pLocked locked or not
012     */
013    public Door(String pDirection, Room pNeighbor, boolean pLocked)
014    {
015        this.alocked = pLocked;
016        this.aDirection=pDirection;
017        this.aNeighbor=pNeighbor;        
018    }
019     
020    /**
021     * Locks this door.
022     */
023    public void lock()
024    {
025        this.alocked = true;
026        return;
027    }
028    
029    /**
030     * Unlocks this door.
031     */
032    public void unlock()
033    {
034        this.alocked = false;
035        return;
036    }
037    
038    /**
039     * 
040     * @return locked or not.
041     */
042    public boolean getLocked()
043    {
044        return this.alocked;
045    }
046    
047    /**
048     * @return aDirection
049     */
050    public String getDirection()
051    {
052        return this.aDirection;
053    }
054    
055    /**
056     * @return aNeighbor
057     */
058    public Room getNeighbor()
059    {
060        return this.aNeighbor;
061    }
062    /**
063     * @param pDirection desared Direction
064     */
065    public void setDirection(String pDirection)
066    {
067        this.aDirection = pDirection;
068    }
069    
070    /**
071     * @param pNeighbor  set room 
072     */
073    public void setNeighbor(Room pNeighbor)
074    {
075        this.aNeighbor = pNeighbor;
076    }
077    
078}
