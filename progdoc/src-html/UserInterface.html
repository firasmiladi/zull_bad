030public class UserInterface implements ActionListener

048    /**
049     * Constru  ct a UserInterface. As a parameter, a Game Engine

052     * 
053     * @param pGameEngine  The GameEngine object implementing the game logic.

055    public UserInterface( final GameEngine pGameEngine )

061    /**
062     * Print out some text into the text area.
063     * @param pText text will be print into the text area
064     */
065    public void print( final String pText )
066    {
067        this.aLog.append( pText );
068        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
069    } // print(.)
070
071    /**
072     * Print out some text into the text area, followed by a line break.
073     * @param pText text will be print into the text area
074     */
075    public void println( final String pText )
076    {
077        this.print( pText + "\n" );
078    } // println(.)
079
080    /**
081     * Show an image file in the interface.
082     * @param pImageName image name
083     */
084    public void showImage( final String pImageName )
085    {
086        String vImagePath = "" + pImageName; // to change the directory
087        URL vImageURL = this.getClass().getClassLoader().getResource("image/"+vImagePath );
088        if ( vImageURL == null )
089            System.out.println( "Image not found : " + vImagePath );
090        else {
091            ImageIcon vIcon = new ImageIcon( vImageURL );
092            this.aImage.setIcon( vIcon );
093            this.aMyFrame.pack();
094        }
095    } // showImage(.)
096
097    /**
098     * Enable or disable input in the input field.
099     * @param pOnOff enable/disable
100     */
101    public void enable( final boolean pOnOff )
102    {
103        this.aEntryField.setEditable( pOnOff ); // enable/disable
104        if ( ! pOnOff ) { // disable
105            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
106            this.aEntryField.removeActionListener( this ); // won't react to entry
107        }
108    } // enable(.)
109

112     * @param pN north

128    /**
129     * Enable or disable buttons if there is no exits.
130     * @param pRoom room
131     */
132    public void boutonvisibility(Room pRoom){
133        this.aButtonN.setVisible(pRoom.getExit("north")!=null);
134        if (pRoom.getExit("north")==null){this.aButtonN.setVisible(false);}
135        else {this.aButtonN.setVisible(true);}
136        if (pRoom.getExit("south")==null){this.aButtonS.setVisible(false);}
137        else {this.aButtonS.setVisible(true);}
138        
139        if (pRoom.getExit("west")==null){this.aButtonW.setVisible(false);}
140        else {this.aButtonW.setVisible(true);}
141        
142        if (pRoom.getExit("east")==null){this.aButtonE.setVisible(false);}
143        else {this.aButtonE.setVisible(true);}
144        
145        if (pRoom.getExit("up")==null){this.aButtonUp.setVisible(false);}
146        else {this.aButtonUp.setVisible(true);}
147        
148        if (pRoom.getExit("down")==null){this.aButtonDown.setVisible(false);}
149        else {this.aButtonDown.setVisible(true);}
150        
151        this.aMyFrame.pack();
152    }

154    /**
155     * Set up graphical user interface.
156     */
157    private void createGUI()
158    {
159        this.aMyFrame = new JFrame( "Firas Miladi" ); // change the title
160        this.aEntryField = new JTextField( 34 );
161        
162        
163        
164        
165        this.aLog = new JTextArea();
166        this.aLog.setEditable( false );
167        JScrollPane vListScroller = new JScrollPane( this.aLog );
168        vListScroller.setPreferredSize( new Dimension(200, 200) );
169        vListScroller.setMinimumSize( new Dimension(100,100) );
170
171        JPanel vPanel = new JPanel();
172        this.aImage = new JLabel();
173
174        vPanel.setLayout( new BorderLayout() ); // ==> only five places
175        vPanel.add( this.aImage, BorderLayout.NORTH );
176        vPanel.add( vListScroller, BorderLayout.CENTER );
177        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
178        
179        this.makeBoutonBar();
180        
181        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.WEST );
182
183        // add some event listeners to some components
184        this.aEntryField.addActionListener( this );
185
186        // to end program when window is closed
187        this.aMyFrame.addWindowListener( new WindowAdapter() {
188                public void windowClosing(WindowEvent e) { System.exit(0); }
189            } );
190        this.aButtonN.setVisible(false);
191        this.aButtonS.setVisible(false);
192        this.aButtonE.setVisible(false);
193        this.aButtonDown.setVisible(false);
194        
195        this.aMyFrame.pack();
196        this.aMyFrame.setVisible( true );
197        this.aEntryField.requestFocus();
198    } // createGUI()
199
200    /**
201     * Actionlistener interface for entry textfield.
202     * @param pE action created by the user
203     */
204    public void actionPerformed( final ActionEvent pE ) 
205    {
206        // no need to check the type of action at the moment
207        // because there is only one possible action (text input) :
208        if(pE.getSource() == this.aButtonN){ this.aEngine.interpretCommand("go north");}
209        else if(pE.getSource() == this.aButtonS){ this.aEngine.interpretCommand("go south");}
210        else if(pE.getSource() == this.aButtonE){ this.aEngine.interpretCommand("go east");}
211        else if(pE.getSource() == this.aButtonW){ this.aEngine.interpretCommand("go west");}
212        else if(pE.getSource() == this.aButtonUp){ this.aEngine.interpretCommand("go up");}
213        else if(pE.getSource() == this.aButtonDown){ this.aEngine.interpretCommand("go down");}
214        else if(pE.getSource() == this.aButtonLook){ this.aEngine.interpretCommand("look");}
215        else if(pE.getSource() == this.aButtonHelp){ this.aEngine.interpretCommand("help");}
216        else if(pE.getSource() == this.aEntryField) {processCommand();}    
217    } // actionPerformed(.)

271    /**
272     * A command has been entered. Read the command and do whatever is 
273     * necessary to process it.
274     */
275    private void processCommand()
276    {
277        String vInput = this.aEntryField.getText();
278        this.aEntryField.setText( "" );
279
280        this.aEngine.interpretCommand( vInput );
281    } // processCommand()
