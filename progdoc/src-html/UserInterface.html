023/**
024 * This class implements a simple graphical user interface with a text entry
025 * area, a text output area and an optional image.
026 * 
027 * @author Michael Kolling
028 * @version 1.0 (Jan 2003) DB edited (2019)
029 */
030public class UserInterface implements ActionListener
031{
032    private GameEngine aEngine;
033    private JFrame     aMyFrame;
034    private JTextField aEntryField;
035    private JTextArea  aLog;
036    private JLabel     aImage;
037    private JPanel     aButton;
038    private JButton     aButtonN;
039    private JButton     aButtonU;
040    private JButton     aButtonD;
041    private JButton     aButtonS;
042    private JButton     aButtonE;
043    private JButton     aButtonW;
044    private JButton     aButtonEat;
045    private JButton     aButtonLook;
046    private JButton     aButtonHelp;
047    /**
048     * Constru  ct a UserInterface. As a parameter, a Game Engine
049     * (an object processing and executing the game commands) is
050     * needed.
051     * 
052     * @param pGameEngine  The GameEngine object implementing the game logic.
053     */
054    public UserInterface( final GameEngine pGameEngine )
055    {
056        this.aEngine = pGameEngine;
057        this.createGUI();
058    } // UserInterface(.)
059
060    /**
061     * Print out some text into the text area.
062     * @param pText text will be print into the text area
063     */
064    public void print( final String pText )
065    {
066        this.aLog.append( pText );
067        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
068    } // print(.)
069
070    /**
071     * Print out some text into the text area, followed by a line break.
072     * @param pText text will be print into the text area
073     */
074    public void println( final String pText )
075    {
076        this.print( pText + "\n" );
077    } // println(.)
078
079    /**
080     * Show an image file in the interface.
081     * @param pImageName image name
082     */
083    public void showImage( final String pImageName )
084    {
085        String vImagePath = "" + pImageName; // to change the directory
086        URL vImageURL = this.getClass().getClassLoader().getResource("image/"+vImagePath );
087        if ( vImageURL == null )
088            System.out.println( "Image not found : " + vImagePath );
089        else {
090            ImageIcon vIcon = new ImageIcon( vImageURL );
091            this.aImage.setIcon( vIcon );
092            this.aMyFrame.pack();
093        }
094    } // showImage(.)
095
096    /**
097     * Enable or disable input in the input field.
098     * @param pOnOff enable/disable
099     */
100    public void enable( final boolean pOnOff )
101    {
102        this.aEntryField.setEditable( pOnOff ); // enable/disable
103        if ( ! pOnOff ) { // disable
104            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
105            this.aEntryField.removeActionListener( this ); // won't react to entry
106        }
107    } // enable(.)
108    
109    /**
110     * Enable or disable buttons if there is no exits.
111     * @param pRoom room
112     */
113    public void boutonvisibility(Room pRoom){
114        this.aButtonN.setVisible(pRoom.getExit("north")!=null);
115        if (pRoom.getExit("north")==null){this.aButtonN.setVisible(false);}
116        else {this.aButtonN.setVisible(true);}
117        if (pRoom.getExit("south")==null){this.aButtonS.setVisible(false);}
118        else {this.aButtonS.setVisible(true);}
119        
120        if (pRoom.getExit("west")==null){this.aButtonW.setVisible(false);}
121        else {this.aButtonW.setVisible(true);}
122        
123        if (pRoom.getExit("east")==null){this.aButtonE.setVisible(false);}
124        else {this.aButtonE.setVisible(true);}
125        
126        this.aMyFrame.pack();
127    }
128    
129    /**
130     * Set up graphical user interface.
131     */
132    private void createGUI()
133    {
134        this.aMyFrame = new JFrame( "Zork" ); // change the title
135        this.aEntryField = new JTextField( 34 );
136        
137        
138        
139        
140        this.aLog = new JTextArea();
141        this.aLog.setEditable( false );
142        JScrollPane vListScroller = new JScrollPane( this.aLog );
143        vListScroller.setPreferredSize( new Dimension(200, 200) );
144        vListScroller.setMinimumSize( new Dimension(100,100) );
145
146        JPanel vPanel = new JPanel();
147        this.aImage = new JLabel();
148
149        vPanel.setLayout( new BorderLayout() ); // ==> only five places
150        vPanel.add( this.aImage, BorderLayout.NORTH );
151        vPanel.add( vListScroller, BorderLayout.CENTER );
152        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
153        
154        this.makeBoutonBar();
155        
156        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.WEST );
157
158        // add some event listeners to some components
159        this.aEntryField.addActionListener( this );
160
161        // to end program when window is closed
162        this.aMyFrame.addWindowListener( new WindowAdapter() {
163                public void windowClosing(WindowEvent e) { System.exit(0); }
164            } );
165        this.aButtonN.setVisible(false);
166        this.aButtonS.setVisible(false);
167        this.aButtonE.setVisible(false);
168        
169        this.aMyFrame.pack();
170        this.aMyFrame.setVisible( true );
171        this.aEntryField.requestFocus();
172    } // createGUI()
173
174    /**
175     * Actionlistener interface for entry textfield.
176     */
177    public void actionPerformed( final ActionEvent pE ) 
178    {
179        // no need to check the type of action at the moment
180        // because there is only one possible action (text input) :
181        if(pE.getSource() == this.aButtonN){ this.aEngine.interpretCommand("go north");}
182        else if(pE.getSource() == this.aButtonS){ this.aEngine.interpretCommand("go south");}
183        else if(pE.getSource() == this.aButtonE){ this.aEngine.interpretCommand("go east");}
184        else if(pE.getSource() == this.aButtonW){ this.aEngine.interpretCommand("go west");}
185        else if(pE.getSource() == this.aButtonEat){ this.aEngine.interpretCommand("eat cookie");}
186        else if(pE.getSource() == this.aButtonLook){ this.aEngine.interpretCommand("look");}
187        else if(pE.getSource() == this.aButtonHelp){ this.aEngine.interpretCommand("help");}
188        else if(pE.getSource() == this.aEntryField) {processCommand();}    
189    } // actionPerformed(.)
190
191    /**
192     * make Bouton Bar.
193     */
194    public void makeBoutonBar()
195    {
196        aButton = new JPanel();
197        aButton.setLayout(new GridLayout(0,1,3,5));
198        
199        this.aButtonN = new JButton("north");        
200        this.aButtonN.addActionListener(this);
201        this.aButtonN.setBackground(Color.RED);
202        
203        this.aButtonS = new JButton("south");
204        this.aButtonS.addActionListener(this);
205        this.aButtonS.setBackground(Color.GREEN);
206        
207        this.aButtonE = new JButton("east");
208        this.aButtonE.addActionListener(this);
209        this.aButtonE.setBackground(Color.YELLOW);
210        
211        this.aButtonW = new JButton("west");
212        this.aButtonW.addActionListener(this);
213        this.aButtonW.setBackground(Color.MAGENTA   );
214       
215        this.aButtonEat = new JButton("eat cookie"); 
216        this.aButtonEat.addActionListener(this);
217        this.aButtonEat.setBackground(Color.CYAN);
218        
219        this.aButtonLook = new JButton("look"); 
220        this.aButtonLook.addActionListener(this);
221        this.aButtonLook.setBackground(Color.WHITE);
222        
223        this.aButtonHelp = new JButton("help"); 
224        this.aButtonHelp.addActionListener(this);
225        this.aButtonHelp.setBackground(Color.ORANGE);
226
227        aButton.add( this.aButtonN);
228        aButton.add( this.aButtonS);
229        aButton.add( this.aButtonE);
230        aButton.add( this.aButtonW);
231        aButton.add( this.aButtonEat);  
232        aButton.add( this.aButtonHelp);
233        aButton.add( this.aButtonLook);
234        
235        this.aMyFrame.getContentPane().add(aButton, BorderLayout.EAST);
236    }
237
238    /**
239     * A command has been entered. Read the command and do whatever is 
240     * necessary to process it.
241     */
242    private void processCommand()
243    {
244        String vInput = this.aEntryField.getText();
245        this.aEntryField.setText( "" );
246
247        this.aEngine.interpretCommand( vInput );
248    } // processCommand()
249} // UserInterface
