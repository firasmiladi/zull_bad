002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * 
006 * This class create the game and initialise its internal map.
007 *
008 * @author  Miladi Firas
009 * 
010 * @version 2021.02.01
011 */
012public class Game
013{
014    private UserInterface aGui;
015    private GameEngine aEngine;
016
017    /**
018     * Create the game and initialise its internal map. Create the inerface and link to it.
019     */
020    public Game() 
021    {
022        this.aEngine = new GameEngine();
023        this.aGui = new UserInterface( this.aEngine );
024        this.aEngine.setGUI( this.aGui );
025    }
026}
