010/**
011 *This class is part of the "World of Zuul" application. 
012 * "World of Zuul" is a very simple, text based adventure game.  
013 * 
014 * This class is used to trait the music in this game .
015 * 
016 * @author Firas miladi + source=https://stackoverflow.com/questions/26305/how-can-i-play-sound-in-java 
017 * @version 30/04/2020                                                  
018 */
019public class SoundEffect
020{
021    private Clip myClip;
022
023    /**
024     * Constructeur par defaut d'objets de la classe SoundEffect
025     * @param pFileName music title
026     */
027    public SoundEffect(final String pFileName)
028    {
029      try {
030          File file = new File(pFileName);
031        Clip myClip = AudioSystem.getClip(); 
032        AudioInputStream sound = AudioSystem.getAudioInputStream(file); 
033        myClip.open(sound);
034      } catch(Exception e){}// SoundEffect()
035    }
036    
037    public void play() {myClip.start();}
038    
039    public void close() {myClip.close();}
040    
041} // SoundEffect
