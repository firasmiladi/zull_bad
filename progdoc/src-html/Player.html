119    /**
120     * add item to your inventory
121     * @param pNoun Item noun
122     * @param pItem Item 
123     */
124    public void addItemInventory (final String pNoun,final Item pItem) { 
125        if(pNoun.equals("gold")){
126            this.aMoney=this.aMoney+2;
127       }
128        else{
129            this.aCatalogue.addItemToTheList(pNoun,pItem); 
130            this.aWeightInventory = this.aWeightInventory+pItem.getWeightItem();
131            this.aMoney = this.aMoney-pItem.getPriceItem();
132       }
133    }

138     */
139    public void removeItemEatableInventory (final String pNounItem) { 
140        Item vItemdrop = this.aCatalogue.getItemList(pNounItem);
141        this.aCatalogue.removeItemFromTheList(pNounItem);
142        this.aWeightInventory= this.aWeightInventory-vItemdrop.getWeightItem();

149    public void removeItemInventory (final String pNounItem) { 
150        Item vItemdrop = this.aCatalogue.getItemList(pNounItem);
151        this.aCatalogue.removeItemFromTheList(pNounItem);
152        this.aWeightInventory= this.aWeightInventory-vItemdrop.getWeightItem();
153        this.aMoney= this.aMoney + vItemdrop.getPriceItem();
154    }

156    /**
157     * @param pCommand item 's noun
158     * @return String of items that has been token .
159     */
160    public String takeString(String pCommand){
161        Item vItemRoom = this.getCurrentRoom().getRoomItemlist().getItemList(pCommand);
162        if (vItemRoom == null) {
163            return "\n there is no such as this thing in this room ";
164            
165        }else {
166            if (pCommand.equals("gold")){
167        
168            this.addItemInventory(pCommand,vItemRoom);
169            this.getCurrentRoom().getRoomItemlist().removeItemFromTheList(pCommand);
170            return "You added a " + vItemRoom.getDescriptionItem() + " to your wallet  ";
171            
172           } 
173           else {
174           this.addItemInventory(pCommand,vItemRoom);
175            this.getCurrentRoom().getRoomItemlist().removeItemFromTheList(pCommand);
176            return "You added a " + vItemRoom.getDescriptionItem() + " to your inventory  ";
177          }
178       }
179    }

185    public String dropString (String pCommand) {
186        Item vItemdrop = this.aCatalogue.getItemList(pCommand);
187        if (vItemdrop == null) {
188            return "\n You don't own this item";
189        }
190        else {
191            this.removeItemInventory(pCommand);
192            this.getCurrentRoom().getRoomItemlist().addItemToTheList(vItemdrop.getNounItem(),vItemdrop);
193            return "You threw a " + vItemdrop.getDescriptionItem() + " from your inventory ";
194        }
195    }

203       Room vRoomLoad = this.getCurrentRoom();

206              if(vItem == null){return "\n" + "You don't have the teleporter with you.";}

215                return "you can't charge this item";
216       }
