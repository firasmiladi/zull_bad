140    /**
141     * add item to your inventory
142     * @param pNoun Item noun
143     * @param pItem Item 
144     */
145    public void addItemInventory (final String pNoun,final Item pItem) { 
146        
147        if(pNoun.equals("gold")){
148            this.aMoney=this.aMoney+2;
149       }
150       else if(pNoun.equals("Alexa")||pNoun.equals("Sarah")){
151            
152           this.aCar.addItemToTheList(pNoun,pItem);
153           
154       }
155        else{
156            this.aCatalogue.addItemToTheList(pNoun,pItem); 
157            this.aWeightInventory = this.aWeightInventory+pItem.getWeightItem();
158            this.aMoney = this.aMoney-pItem.getPriceItem();
159       }
160    }

202    /**
203     * @param pCommand item 's noun
204     * @return String of items that has been token .
205     */
206    public String takeString(String pCommand){
207        Item vItemRoom = this.getCurrentRoom().getRoomItemlist().getItemList(pCommand);
208        if (vItemRoom == null) {
209            return "\n there is no such as this thing in this room ";
210            
211        }else{
212            this.addItemInventory(pCommand,vItemRoom);
213            this.getCurrentRoom().getRoomItemlist().removeItemFromTheList(pCommand);
214            if (pCommand.equals("gold")){
215                return "You added a " + vItemRoom.getDescriptionItem() + " to your wallet  ";
216            
217           }
218           else{
219            return "You added a " + vItemRoom.getDescriptionItem() + " to your inventory  ";
220          }
221       }
222    }

226     * @return String of items that has been drop in .
227     */
228    public String dropString (String pCommand) {
229        Item vItemdrop = this.aCatalogue.getItemList(pCommand);
230        if (vItemdrop == null) {
231            return "\n You don't own this item";
232        }
233        else {
234            this.removeItemInventory(pCommand);
235            this.getCurrentRoom().getRoomItemlist().addItemToTheList(vItemdrop.getNounItem(),vItemdrop);
236            return "You threw a " + vItemdrop.getDescriptionItem() + " from your inventory ";
237        }
238    }
